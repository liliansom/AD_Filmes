# Para verificar as informações da tabela
SELECT *
FROM netflix


# Para verificar a incidência das Classificações dos filmes
SELECT TRIM(value) AS listed_in_item, COUNT(*) AS count
FROM netflix
CROSS APPLY STRING_SPLIT(listed_in, ',')
GROUP BY TRIM(value)
ORDER BY count DESC


# Para verificar a incidência dos Diretores dos filmes
SELECT TRIM(value) AS directors_item, COUNT(*) AS count
FROM netflix
CROSS APPLY STRING_SPLIT(director, ',')
GROUP BY TRIM(value)
ORDER BY count DESC


# Para verificar a incidência dos Países dos filmes
SELECT TRIM(value) AS directors_item, COUNT(*) AS count
FROM netflix
CROSS APPLY STRING_SPLIT(country, ',')
GROUP BY TRIM(value)
ORDER BY count DESC
